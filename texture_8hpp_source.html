291inline texture_2d::texture_2d(std::span<const float> data, resolution res,
292 texture_color color, texture_filter filter, tex_samples samples, bool generate_mipmap) noexcept
293 : m_color { color }
294 , m_filter { filter }
295 , m_resolution { res }
296 , m_antialias { (samples == tex_samples::MSAA_X1) ? texture_antialias { GL_TEXTURE_2D, samples } : texture_antialias { GL_TEXTURE_2D_MULTISAMPLE, samples } }
297{
298 glGenTextures(1, &m_id);
299 glBindTexture(m_antialias.type, m_id);
300
301 if (m_antialias.type == GL_TEXTURE_2D) {
302 glTexParameteri(m_antialias.type, GL_TEXTURE_MIN_FILTER, generate_mipmap ? to_mipmap(filter.min_filter) : filter.min_filter);
303 glTexParameteri(m_antialias.type, GL_TEXTURE_MAG_FILTER, filter.mag_filter);
304 glTexParameteri(m_antialias.type, GL_TEXTURE_WRAP_S, filter.clamping);
305 glTexParameteri(m_antialias.type, GL_TEXTURE_WRAP_T, filter.clamping);
306
307 glTexImage2D(m_antialias.type, 0, color.internal_format, m_resolution.width, m_resolution.height, 0, color.format, color.datatype, data.data());
308 } else if (m_antialias.type == GL_TEXTURE_2D_MULTISAMPLE) {
309 glTexImage2DMultisample(m_antialias.type, m_antialias.samples, color.internal_format,
310 m_resolution.width, m_resolution.height, GL_TRUE);
311 }
312
313 if (generate_mipmap) {
314 glGenerateMipmap(m_antialias.type);
315 }
316
317 glBindTexture(m_antialias.type, 0);
318}
