232inline void framebuffer::set_renderbuffer(resolution res, fbo_attachment attachment, tex_samples samples)
233{
234 if (attachment != fbo_attachment::NONE) {
235
236 renderbuffer::attachment_type type {};
237
238 switch (attachment) {
239 case fbo_attachment::ATTACH_DEPTH_BUFFER:
240 type = renderbuffer::attachment_type::depth;
241 break;
242 case fbo_attachment::ATTACH_STENCIL_BUFFER:
243 type = renderbuffer::attachment_type::stencil;
244 break;
245 case fbo_attachment::ATTACH_DEPTH_STENCIL_BUFFER:
246 type = renderbuffer::attachment_type::depth_stencil;
247 break;
248 default:
249#ifdef STAPLEGL_DEBUG
250 std::fprintf(stderr, STAPLEGL_LINEINFO "invalid attachment enum %d for renderbuffer\n", static_cast<std::uint32_t>(attachment));
251#endif
252 std::terminate(); // crash and burn, this is unrecoverable
253 }
254
255 m_attachment = attachment; // in some cases this is a redundant operation, fair enough.
256 m_renderbuffer.emplace(res, type, samples);
257
258 glFramebufferRenderbuffer(GL_FRAMEBUFFER, static_cast<std::uint32_t>(type), GL_RENDERBUFFER, m_renderbuffer->id());
259 } else {
260 // unbind the old framebuffer, if present
261
262 if (m_attachment == fbo_attachment::NONE) {
263 return; // nothing to unbind
264 }
265
266 renderbuffer::attachment_type type {};
267
268 switch (m_attachment) {
269 case fbo_attachment::ATTACH_DEPTH_BUFFER:
270 type = renderbuffer::attachment_type::depth;
271 break;
272 case fbo_attachment::ATTACH_STENCIL_BUFFER:
273 type = renderbuffer::attachment_type::stencil;
274 break;
275 case fbo_attachment::ATTACH_DEPTH_STENCIL_BUFFER:
276 type = renderbuffer::attachment_type::depth_stencil;
277 break;
278 default:
279 break;
280 }
281
282 m_attachment = fbo_attachment::NONE;
283
284 glFramebufferRenderbuffer(GL_FRAMEBUFFER,
285 static_cast<std::uint32_t>(type),
286 GL_RENDERBUFFER, 0);
287 }
288}
